import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class SistemaRH {
    public static void main(String[] args) {
        // Criando scanner para entrada de dados do usuário
        public static void main(String[] args) {
        // Criando scanner para entrada de dados do usuário
        Scanner scanner = new Scanner(System.in);
        
        // Lista que armazenará todos os funcionários (polimorfismo - pode conter diferentes tipos)
        List<Funcionario> funcionarios = new ArrayList<>();
        
        System.out.println("=== SISTEMA DE RH - CADASTRO DE FUNCIONÁRIOS ===");
        System.out.println("Você pode cadastrar até 10 funcionários.");
        
        // Loop para cadastrar até 10 funcionários
        for (int i = 0; i < 10; i++) {
            System.out.println("\n--- Cadastro do Funcionário " + (i + 1) + " ---");
            
            // Pergunta se deseja cadastrar mais um funcionário
            System.out.print("Deseja cadastrar um funcionário? (s/n): ");
            String continuar = scanner.nextLine().toLowerCase();
            
            // Se não quiser continuar, quebra o loop
            if (!continuar.equals("s") && !continuar.equals("sim")) {
                break;
            }
            
            // Coletando dados comuns a todos os funcionários
            System.out.print("Nome: ");
            String nome = scanner.nextLine();
            
            System.out.print("CPF: ");
            String cpf = scanner.nextLine();
            
            System.out.print("Endereço: ");
            String endereco = scanner.nextLine();
            
            System.out.print("Telefone: ");
            String telefone = scanner.nextLine();
            
            System.out.print("Setor: ");
            String setor = scanner.nextLine();
            
            // Perguntando o tipo de funcionário
            System.out.print("Tipo de funcionário (1-Assalariado / 2-Horista): ");
            int tipo = scanner.nextInt();
            scanner.nextLine(); // Limpa o buffer do scanner
            
            // Criando funcionário baseado no tipo escolhido (polimorfismo)
            if (tipo == 1) {
                // Coletando dados específicos do assalariado
                System.out.print("Salário mensal: R$ ");
                double salarioMensal = scanner.nextDouble();
                scanner.nextLine(); // Limpa o buffer
                
                // Criando objeto FuncionarioAssalariado e adicionando à lista
                FuncionarioAssalariado funcionario = new FuncionarioAssalariado(
                    nome, cpf, endereco, telefone, setor, salarioMensal);
                funcionarios.add(funcionario);
                
            } else if (tipo == 2) {
                // Coletando dados específicos do horista
                System.out.print("Horas trabalhadas: ");
                double horasTrabalhadas = scanner.nextDouble();
                
                System.out.print("Valor por hora: R$ ");
                double valorHora = scanner.nextDouble();
                scanner.nextLine(); // Limpa o buffer
                
                // Criando objeto FuncionarioHorista e adicionando à lista
                FuncionarioHorista funcionario = new FuncionarioHorista(
                    nome, cpf, endereco, telefone, setor, horasTrabalhadas, valorHora);
                funcionarios.add(funcionario);
                
            } else {
                // Tipo inválido - decrementa i para repetir o cadastro
                System.out.println("Tipo inválido! Tente novamente.");
                i--;
                continue;
            }
            
            System.out.println("Funcionário cadastrado com sucesso!");
        }
        
        // Verificando se há funcionários cadastrados
        if (funcionarios.isEmpty()) {
            System.out.println("Nenhum funcionário foi cadastrado.");
            scanner.close();
            return;
        }
        
        // Mostrando dados e pagamentos de todos os funcionários
        System.out.println("\n=== DADOS DOS FUNCIONÁRIOS CADASTRADOS ===");
        for (int i = 0; i < funcionarios.size(); i++) {
            System.out.println("\n--- Funcionário " + (i + 1) + " ---");
            // Polimorfismo em ação - chama o método correto baseado no tipo real do objeto
            funcionarios.get(i).mostrarDados();
        }
        
        // Aplicando aumento geral
        System.out.print("\nInforme o percentual de aumento geral para todos os funcionários: ");
        double percentualAumento = scanner.nextDouble();
        
        // Aplicando aumento para todos os funcionários (polimorfismo)
        for (Funcionario funcionario : funcionarios) {
            // Cada tipo de funcionário aplica o aumento de forma específica
            funcionario.aplicarAumento(percentualAumento);
        }
        
        // Mostrando os pagamentos após o aumento
        System.out.println("\n=== PAGAMENTOS APÓS AUMENTO DE " + percentualAumento + "% ===");
        for (int i = 0; i < funcionarios.size(); i++) {
            System.out.println("\n--- Funcionário " + (i + 1) + " ---");
            System.out.println("Nome: " + funcionarios.get(i).getNome());
            System.out.println("Tipo: " + 
                (funcionarios.get(i) instanceof FuncionarioAssalariado ? "Assalariado" : "Horista"));
            
            // Polimorfismo - o método correto é chamado baseado no tipo real
            System.out.println("Novo Pagamento: R$ " + 
                String.format("%.2f", funcionarios.get(i).calcularSalario()));
        }
        
        // Fechando o scanner
        scanner.close();
        
        System.out.println("\nSistema finalizado!");
    }
}
